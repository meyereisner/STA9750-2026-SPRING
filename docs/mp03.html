state_to_state_migration_2024 <- function(){
    
    data_dir <- file.path("data", "mp03")
    if(!dir.exists(data_dir)){
        dir.create(data_dir, showWarnings = FALSE, recursive = TRUE)
    }
    
    file_url <- paste0("https://www2.census.gov/programs-surveys/demo/",
                        "tables/geographic-mobility/2024/state-to-state-migration/",
                        "State_to_State_Migration_Table_2024_T13.xlsx")
    file_path <- file.path(data_dir, "State_to_State_Migration_Table_2024.xlsx")
    
    if(!file.exists(file_path)){
        message(glue("Downloading 2024 state-to-state migration file from {file_url}"))
        download.file(file_url, destfile = file_path, quiet = TRUE, mode = "wb")
    }
    
    # off-diagonal flows from the "Table" sheet
    flows_table <- read_excel(
            file_path,
            sheet = "Table",
            skip = 8,
            col_names = c("state_current", "state_1y", "estimate", "moe")
        ) |>
        mutate(state_current = str_trim(state_current),
               state_1y = str_trim(state_1y)) |>
        filter(state_current %in% state.name,
               state_1y %in% state.name) |>
        filter(estimate != "X") |>
        mutate(estimate = if_else(estimate == "N", "0", estimate),
               estimate = as.integer(estimate)) |>
        # convert full names to two-letter abbreviations
        mutate(state_current = state.abb[match(state_current, state.name)],
               state_1y = state.abb[match(state_1y,      state.name)]) |>
        select(state_current, state_1y, population = estimate) |>
        mutate(year_current = 2024L, year_1y = 2023L)
    
    # stayed in state diagonal from the Supplemental sheet:
    # col 4 = same residence a year ago, col 6 = moved within same state
    flows_stayed <- read_excel(
            file_path,
            sheet = "Supplemental - Current Res",
            skip = 8,
            col_names = FALSE
        ) |>
        select(state_current = 1,
               same_residence = 4,
               mover_in_state = 6) |>
        mutate(state_current = str_trim(state_current)) |>
        filter(state_current %in% state.name) |>
        mutate(population = as.integer(same_residence) +
                            as.integer(mover_in_state),
               state_1y   = state_current) |>
        mutate(state_current = state.abb[match(state_current, state.name)],
               state_1y = state.abb[match(state_1y,      state.name)],
               year_current = 2024L,
               year_1y = 2023L) |>
        select(state_current, state_1y, population, year_current, year_1y)
    
    bind_rows(flows_table, flows_stayed) |>
        arrange(state_current, state_1y)
}

state_to_state_migration_2023 <- function(){

    data_dir <- file.path("data", "mp03")
    if(!dir.exists(data_dir)){
        dir.create(data_dir, showWarnings = FALSE, recursive = TRUE)
    }

    file_url  <- paste0("https://www2.census.gov/programs-surveys/demo/",
                        "tables/geographic-mobility/2023/state-to-state-migration/",
                        "State_to_State_Migration_Table_2023_T13.xlsx")
    file_path <- file.path(data_dir, "State_to_State_Migration_Table_2023.xlsx")
    if(!file.exists(file_path)){
        message("Downloading 2023 state-to-state migration file")
        download.file(file_url, destfile = file_path, quiet = TRUE, mode = "wb")
    }

    states_with_dc <- c(state.name[1:8], "District of Columbia", state.name[9:50])

    # stayed-in-state diagonal (col D same residence + col F moved within state)
    stayed <- read_excel(
            file_path,
            range = "A12:F62",
            col_names = c("state_current", "pop_est", "pop_moe",
                          "same_res", "same_res_moe", "same_state")
        ) |>
        mutate(state_current = str_trim(state_current)) |>
        filter(state_current %in% state.name) |>
        mutate(population = as.integer(same_res) + as.integer(same_state),
               state_1y = state_current,
               year_current = 2023L,
               year_1y = 2022L) |>
        mutate(state_current = state.abb[match(state_current, state.name)],
               state_1y = state.abb[match(state_1y,state.name)]) |>
        select(state_current, state_1y, population, year_current, year_1y)

    # between-state flow matrix J:DG; interleave state names with NA so the estimate columns are named and MOE columns get dropped by any_of()
    between <- read_excel(
            file_path,
            range = "J12:DG62",
            col_names = vec_interleave(states_with_dc, NA)
        ) |>
        select(any_of(states_with_dc)) |>
        mutate(state_current = states_with_dc) |>
        pivot_longer(cols = -state_current,
                     names_to = "state_1y",
                     values_to = "population") |>
        filter(state_current != "District of Columbia",
               state_1y != "District of Columbia") |>
        filter(state_current != state_1y) |>
        mutate(population = as.integer(population),
               year_current = 2023L,
               year_1y = 2022L) |>
        mutate(state_current = state.abb[match(state_current, state.name)],
               state_1y = state.abb[match(state_1y, state.name)]) |>
        select(state_current, state_1y, population, year_current, year_1y)

    bind_rows(stayed, between) |>
        arrange(state_current, state_1y)
}

metro_to_metro_migration <- function(){

    data_dir <- file.path("data", "mp03")
    if(!dir.exists(data_dir)){
        dir.create(data_dir, showWarnings = FALSE, recursive = TRUE)
    }

    api_path  <- paste0(
        "https://api.census.gov/data/2020/acs/flows",
        "?get=MOVEDIN,MOVEDOUT,FULL1_NAME,FULL2_NAME",
        "&for=", URLencode(
            "metropolitan statistical area/micropolitan statistical area:*")
    )
    json_path <- file.path(data_dir, "metro_to_metro_flows_2016_2020.json")

    if(!file.exists(json_path)){
        message("Downloading 2016-2020 metro-to-metro migration flows from Census API")
        download.file(api_path, destfile = json_path, quiet = TRUE, mode = "wb")
    }

    move_row_to_colnames <- function(X, row = 1){
        X <- as_tibble(X)
        Xrow <- X[row, ]
        X <- X[-row, ]
        colnames(X) <- Xrow
        readr::type_convert(X)
    }

    pull_state_from_metro <- function(metro_name){
        str_extract(metro_name,
                    ".*, (\\S{2})[-[:alpha:]]* Metro Area",
                    group = 1)

    read_json(json_path, simplifyVector = TRUE) |>
        move_row_to_colnames() |>
        select(metro_current = FULL1_NAME,
               metro_1y = FULL2_NAME,
               moved_in = MOVEDIN,
               moved_out = MOVEDOUT) |>
        mutate(state_current = pull_state_from_metro(metro_current),
               state_1y = pull_state_from_metro(metro_1y))

# Question 10: Are there any metro areas that have a particularly high connection to your state’s largest metro?

# denominators: for each metro X, total inflow from and outflow to all partners
metro_denoms <- metro_flows |>
    filter(metro_current != metro_1y) |>
    group_by(metro_current) |>
    summarise(total_in  = sum(moved_in,  na.rm = TRUE),
              total_out = sum(moved_out, na.rm = TRUE),
              .groups = "drop")

# numerators: for each metro X, flow FROM NYC into X and flow INTO NYC from X.
# The metro flows table reports each ordered pair once, so depending on whether
# NYC shows up as `metro_current` or `metro_1y`, MOVEDIN/MOVEDOUT mean different things:
#   - metro_current == NYC: MOVEDIN = NYC <- other; MOVEDOUT = NYC -> other
#   - metro_1y      == NYC: MOVEDIN = X  <- NYC;   MOVEDOUT = X  -> NYC
nyc_edges <- metro_flows |>
    filter(metro_current != metro_1y,
           (metro_current == NYC_METRO) | (metro_1y == NYC_METRO)) |>
    mutate(other_metro = if_else(metro_current == NYC_METRO, metro_1y, metro_current),
           nyc_to_other = case_when(
               metro_current == NYC_METRO ~ moved_out,
               metro_1y == NYC_METRO ~ moved_in
           ),
           other_to_nyc = case_when(
               metro_current == NYC_METRO ~ moved_in,
               metro_1y == NYC_METRO ~ moved_out
           )) |>
    group_by(other_metro) |>
    summarise(nyc_to_other = sum(nyc_to_other, na.rm = TRUE),
              other_to_nyc = sum(other_to_nyc, na.rm = TRUE),
              .groups = "drop")

connection <- nyc_edges |>
    left_join(metro_denoms, by = c("other_metro" = "metro_current")) |>
    mutate(share_of_inflow = nyc_to_other / total_in) |>
    filter(!is.na(str_extract(other_metro, ", \\S{2}[-[:alpha:]]* Metro Area")),
           total_in  >= 1000,
           total_out >= 1000)

huntington_hill <- function(pops, N = 435){
    # pops: named numeric vector (names = state names)
    # Returns a named integer vector of seats.
    tibble(state = names(pops), population = pops) |>
        expand_grid(cd = seq(0, 100)) |>
        mutate(hh_den    = sqrt(cd * (cd + 1)),
               hh_weight = population / hh_den) |>
        slice_max(hh_weight, n = N) |>
        count(state, name = "seats")
}

Memo to the Governor MEMO To: Governor of New York From: Office of Policy Analysis Re: Advertising Strategy — “Return to New York” Campaign for 2030 Reapportionment The problem, in one paragraph Our population model projects New York at 22.07 million residents in 2030, which produces 24 congressional seats in the 2032 apportionment — a loss of two seats from the 26 we hold today. Under the Huntington-Hill method, the single closest seat in the 435-seat queue belongs to Georgia; the margin is roughly 14,000 priority points. In concrete terms, New York needs to add approximately 350,000 residents beyond our current trajectory to flip that seat back to us and return to 25 representatives. That is the size of the target. Where to advertise The data supports three concentric circles of targeting, in descending political value per dollar spent. Tier 1 — The NY diaspora. The single highest-impact campaign is targeting former New Yorkers who left in the last decade and have retained tax, family, or cultural ties to the state. Our flow data shows the largest destinations of NY out-migrants are Florida, New Jersey, California, Pennsylvania, and Texas — in that order. Florida alone absorbs roughly 65,000 New Yorkers per year. Even a 15% “return rate” among recent movers would meaningfully close our gap. Ad buys should concentrate on the Miami, Tampa, Orlando, and Palm Beach metros; Los Angeles, San Francisco, Dallas, Austin, Houston, and Philadelphia should follow. These are former New Yorkers — the pitch doesn’t need to convince them the state is worth living in; they already know. It needs to remove the specific frictions that made them leave. Tier 2 — High-value inbound flows. Our top inbound feeder states (NJ, PA, CA, FL, CT) already send substantial numbers to New York every year. Incremental advertising here reinforces an existing pattern at low marginal cost. Focus on the Philadelphia, Boston, Bridgeport-Stamford, and Los Angeles metros — concentrated population centers where brand advertising reaches the largest pool of prospective movers. Tier 3 — Vulnerable small states. The professor’s example of Montana is instructive, and the 2030 data confirms it: Delaware, Rhode Island, Montana, and West Virginia each hold a 2nd congressional seat whose priority is narrow enough that ~30,000 residents relocating to New York would simultaneously give us a 25th seat and cost the donor state its 2nd. This is the rare zero-sum play where our gain is directly tied to another state’s loss, which makes it politically sensitive — the campaign in these states should be subtle, emphasizing career opportunity rather than any implicit “come drain your home state” messaging. How much migration to induce To recover one seat (24 → 25), we need to attract approximately 350,000 additional net residents above our current trajectory by the 2030 census. To recover both lost seats (24 → 26), we need roughly 1.35 million additional residents — a much heavier lift and, realistically, likely beyond what any advertising campaign alone can achieve within five years. A realistic operational target is therefore one additional seat. Of the 350,000 needed arrivals, a reasonable split across the three tiers is:

~200,000 from Tier 1 (diaspora return): 3-4% of the NY-born population currently living in FL, CA, TX, and the Northeast ~100,000 from Tier 2 (pattern reinforcement): a 10-15% lift on existing inbound flows from NJ, PA, CA, CT ~50,000 from Tier 3 (vulnerable-state poaching): career-pipeline recruiting in small-state professional markets

These numbers are aggressive but not fantastical. The national annual interstate migration volume into New York is already ~280,000 people; we are asking the campaign to move that number upward by roughly 20-25% per year for five years. Campaign positioning and slogan candidates The tonal requirement is honesty. New Yorkers who left are familiar with the real trade-offs — cost, weather, density — and will tune out any pitch that denies them. The strongest campaigns lean into New York’s genuine differentiators: density of opportunity, cultural depth, public infrastructure, and (now) a competitive set of state-level tax and family incentives that the administration should package alongside the advertising buy.
