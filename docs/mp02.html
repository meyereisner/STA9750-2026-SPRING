Introduction

In this project, we explore the American Time Use Survey (ATUS), a large scale annual survey conducted by the Bureau of Labor Statistics that captures detailed time use diaries from a nationally representative sample of Americans. Each respondent reports how they spent a single 24 hour day, broken down by activity, location, and duration. The dataset goes from 2003 to 2024 and includes over 250,000 respondents along with demographic information such as age, marital status, employment, and household composition.

Our analysis proceeds in several stages. We begin by loading and cleaning the data, adding interpretable demographic variables to prepare the dataset for analysis. We then answer a series of exploratory questions covering topics like retirement rates, sleep patterns, childcare time, and sports viewership. Finally, we use the data to build time-use profiles for three target demographics as part of a market research exercise for a hypothetical time-tracking app.

load_atus_data <- function(file_base = c("resp", "rost", "act")){
    if(!dir.exists(file.path("data", "mp02"))){
        dir.create(file.path("data", "mp02"), showWarnings=FALSE, recursive=TRUE)
    }
    
    file_base <- match.arg(file_base)
    
    file_name_out <- file.path("data", 
                               "mp02", 
                               glue("atus{file_base}_0324.dat"))
        
    if(!file.exists(file_name_out)){
        
        url_end <- glue("atus{file_base}-0324.zip")
        
        temp_zip <- tempfile(fileext=".zip")
        temp_dir <- tools::file_path_sans_ext(temp_zip)
        
        request("https://www.bls.gov") |>
            req_url_path("tus", "datafiles", url_end) |>
            req_headers(`User-Agent` = "Mozilla/5.0 (Macintosh; Intel Mac OS X 10.15; rv:149.0) Gecko/20100101 Firefox/149.0") |>
            req_perform(temp_zip)
    
        unzip(temp_zip, exdir = temp_dir)
        
        temp_file <- file.path(temp_dir, basename(file_name_out))
        
        file.copy(temp_file, file_name_out)
    }
    
    file <- read_csv(file_name_out, 
                     show_col_types = FALSE)
    
    switch(file_base, 
           resp = file |> 
                  rename(survey_weight = TUFNWGTP, 
                         time_alone = TRTALONE, 
                         survey_year = TUYEAR) |>
                  mutate(
                    # Task 3: Number of own household children under 18
                    n_children_under18 = TRYHHCHILD,
                    
                    # Retirement status from TERET1: 2 = retired, 1 = not retired, negative = not applicable
                    retired = if_else(TERET1 == 2, TRUE, FALSE),
                    
                    # Task 3: Married status based on spouse presence
                    # TRSPPRES: 1 = spouse present, 2 = unmarried partner, 3 = no spouse/partner
                    married = case_match(TRSPPRES,
                      1 ~ "Yes",
                      2 ~ "No",
                      3 ~ "No"
                    ),
                    
                    # Task 3 and task 4: Employment status 
                    employment_status = case_match(TELFS,
                      1 ~ "Employed - At Work",
                      2 ~ "Employed - Absent",
                      3 ~ "Unemployed - On Layoff",
                      4 ~ "Unemployed - Looking",
                      5 ~ "Not in Labor Force"
                    ),
                    
                    # Task 3: College enrollment
                    # TESCHLVL: -1 = not enrolled, 1 = high school, 2 = college/university
                    # We need enrolled AND at college level (not HS), so TESCHLVL == 2
                    college_enrolled = if_else(TESCHLVL == 2, "Yes", "No"),
                    
                    # Task 5:
                    line_no = TULINENO,
                    hourly_earnings = TRERNHLY,
                    
                    # Task 6:
                    
                    # Survey month (1-12)
                    survey_month = TUMONTH,
                    
                    # Industry code (for identifying education workers)
                    industry_code = TEIO1ICD,
                  ),
           # rost = file |>
                  # mutate(sex = case_match(TESEX, 1 ~ "M", 2 ~ "F")), 
           
           rost = file |>
                  mutate(sex = case_match(TESEX, 1 ~ "M", 2 ~ "F"),
                         age = TEAGE,
                         line_no = TULINENO),
           act = file |>
                mutate(activity_n_of_day = TUACTIVITY_N, 
                       time_spent_min = TUACTDUR24, 
                       start_time = TUSTARTTIM,
                       stop_time = TUSTOPTIME, 
                       level1_code = paste0(TRTIER1P, "0000"), 
                       level2_code = paste0(TRTIER2P, "00"),
                       level3_code = TRCODEP,
                       
                       # Task 3: Location of activity
                       activity_location = case_match(TEWHERE,
                         1  ~ "Respondent's Home",
                         2  ~ "Respondent's Workplace",
                         3  ~ "Someone Else's Home",
                         4  ~ "Restaurant or Bar",
                         5  ~ "Place of Worship",

compute_time_plot <- function(demo_acts, demo_resp, label, target_min) {
  pp <- demo_acts |>
    group_by(participant_id) |>
    summarise(total_min = sum(time_spent_min), .groups = "drop") |>
    right_join(demo_resp |> select(participant_id, survey_weight), by = "participant_id") |>
    mutate(total_min = replace_na(total_min, 0))
  
  ord <- pp |> arrange(total_min)
  cw <- cumsum(ord$survey_weight) / sum(ord$survey_weight)
  
  tibble(
    activity = label,
    percentile = c("25th", "50th", "75th"),
    minutes = c(
      ord$total_min[which(cw >= 0.25)[1]],
      ord$total_min[which(cw >= 0.50)[1]],
      ord$total_min[which(cw >= 0.75)[1]]
    ),
    target = target_min
  )
}

load_atus_data <- function(file_base = c("resp", "rost", "act")){
    if(!dir.exists(file.path("data", "mp02"))){

        temp_zip <- tempfile(fileext=".zip")
        temp_dir <- tools::file_path_sans_ext(temp_zip)
        
        request("https://www.bls.gov") |>
            req_url_path("tus", "datafiles", url_end) |>
            req_headers(`User-Agent` = "Mozilla/5.0 (Macintosh; Intel Mac OS X 10.15; rv:149.0) Gecko/20100101 Firefox/149.0") |>
            req_perform(temp_zip)
    
        unzip(temp_zip, exdir = temp_dir)
        
        temp_file <- file.path(temp_dir, basename(file_name_out))
        
        file.copy(temp_file, file_name_out)
    }
    
    file <- read_csv(file_name_out, 
                     show_col_types = FALSE)
    
    switch(file_base, 
           resp = file |> 
                  rename(survey_weight = TUFNWGTP, 
                         time_alone = TRTALONE,

                    married = case_match(TRSPPRES,
                      1 ~ "Yes",

# Helper: compute weighted percentiles and return plottable data frame
compute_time_plot <- function(demo_acts, demo_resp, label, target_min) {
  pp <- demo_acts |>
    group_by(participant_id) |>
    summarise(total_min = sum(time_spent_min), .groups = "drop") |>
    right_join(demo_resp |> select(participant_id, survey_weight), by = "participant_id") |>
    mutate(total_min = replace_na(total_min, 0))
  
  ord <- pp |> arrange(total_min)
  cw <- cumsum(ord$survey_weight) / sum(ord$survey_weight)
  
  tibble(
    activity = label,
    percentile = c("25th", "50th", "75th"),
    minutes = c(
      ord$total_min[which(cw >= 0.25)[1]],
      ord$total_min[which(cw >= 0.50)[1]],
      ord$total_min[which(cw >= 0.75)[1]]
    ),
    target = target_min
  )
}
