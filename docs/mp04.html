Introduction

This project builds a statistical forecast for Team USA’s medal count at the 2028 Los Angeles Summer Olympics. The forecast combines three factors: USA’s baseline historical performance, the host nation effect (home field advantage), and the new sport effect (host countries often do well in newly added events). LA 2028 is bringing back six sports: baseball, softball, lacrosse, cricket, flag football, and squash which will add 22 new medal events to the program.

The data comes from Olympedia, scraped using httr2 and rvest. After cleaning the data and standardizing country codes against ISO references, the project moves through exploratory data analysis, formal statistical inference using the infer package to estimate confidence intervals for each of the three factors, and a Monte Carlo simulation that combines those intervals into a final forecast.

slow_download <- purrr::slowly(download.file, purrr::rate_delay(pause = 5))

read_olympedia <- function(url, ...) {
  if (stringr::str_starts(url, "/")) url <- stringr::str_remove(url, "/")
  cache_dir <- fs::path("data", "mp04", "olympedia_cache")
  if (!fs::dir_exists(cache_dir)) fs::dir_create(cache_dir, recurse = TRUE)
  cache_path <- fs::path(cache_dir,
                         url |> as.character() |> stringr::str_replace_all("/", "_"))
  if (!fs::file_exists(cache_path)) {
    src_url <- paste0("https://www.olympedia.org/", url)
    slow_download(src_url, cache_path, mode = "wb")
  }
  rvest::read_html(cache_path)
}

parse_editions_table <- function(table_node, season) {
  if (is.na(table_node)) return(tibble())
  rows <- table_node |> html_elements("tr")

  tibble(
    season      = season,
    olympiad_rn = rows |> html_element("td:nth-of-type(1)") |> html_text2() |> str_trim(),
    edition_id  = rows |> html_element("td:nth-of-type(1) a") |> html_attr("href") |>
                    str_extract("\\d+$") |> as.integer(),
    host_noc    = rows |> html_element("td img") |> html_attr("src") |>
                    str_extract("[A-Z]{3}(?=\\.(png|svg))"),
    year        = rows |> html_element("td:nth-of-type(2)") |> html_text2() |>
                    str_extract("\\b(18|19|20)\\d{2}\\b") |> as.integer()
  ) |>
    mutate(olympiad = suppressWarnings(as.integer(as.roman(olympiad_rn)))) |>
    filter(!is.na(edition_id), !is.na(host_noc), !is.na(year))
}

get_disciplines <- function(edition_id) {
  page <- read_olympedia(paste0("editions/", edition_id))
  links <- page |> html_elements(paste0("a[href*='/editions/", edition_id, "/sports/']"))
  tibble(
    edition_id      = edition_id,
    discipline_code = links |> html_attr("href") |> str_extract("[^/]+$"),
    discipline_name = links |> html_text2() |> str_trim()
  ) |>
    filter(!is.na(discipline_code), discipline_code != "", str_length(discipline_code) <= 5,
           discipline_name != "") |>
    distinct()
}

disciplines <- editions |>

# Task 1 scrape-medal-tables
get_medal_table <- function(edition_id, discipline_code) {
  page <- read_olympedia(paste0("editions/", edition_id, "/sports/", discipline_code))
  tbl_node <- page |>
    html_element(xpath = "//*[self::h2 or self::h3][normalize-space()='Medal table']/following::table[1]")
  if (is.na(tbl_node)) return(tibble())

  rows <- tbl_node |> html_elements("tr")
  noc <- rows |> html_element("td:nth-of-type(2) a") |> html_attr("href") |>
    str_extract("[A-Z]{3}$")

  raw <- html_table(tbl_node, header = FALSE)
  tibble(
    noc    = noc,
    gold   = suppressWarnings(as.integer(raw[[3]])),
    silver = suppressWarnings(as.integer(raw[[4]])),
    bronze = suppressWarnings(as.integer(raw[[5]])),
    total  = suppressWarnings(as.integer(raw[[6]]))
  ) |>
    filter(!is.na(noc), !is.na(gold))
}

# Task 3 eda-4 home field advantage

# For each Summer host, compute gold at host edition and at adjacent editions
summer_editions <- olympics_clean |>
  filter(season == "Summer") |>
  distinct(year, host_noc) |>
  arrange(year)

country_year_gold <- olympics_clean |>
  filter(season == "Summer") |>
  group_by(noc, year) |>
  summarise(gold = sum(gold), .groups = "drop")

host_effect <- summer_editions |>
  rename(host_year = year) |>
  rowwise() |>
  mutate(
    host_gold = country_year_gold |>
      filter(noc == host_noc, year == host_year) |> pull(gold) |>
      (\(x) if (length(x)) x else NA_integer_)(),
    prior_gold = country_year_gold |>
      filter(noc == host_noc, year < host_year) |>
      slice_max(year, n = 1) |> pull(gold) |>
      (\(x) if (length(x)) x else NA_integer_)(),
    next_gold = country_year_gold |>
      filter(noc == host_noc, year > host_year) |>
      slice_min(year, n = 1) |> pull(gold) |>
      (\(x) if (length(x)) x else NA_integer_)()
  ) |>
  ungroup() |>
  mutate(neighbor_avg = rowMeans(cbind(prior_gold, next_gold), na.rm = TRUE),
         premium = host_gold - neighbor_avg) |>
  filter(!is.nan(neighbor_avg), !is.na(host_gold)) |>
  rename(year = host_year)

host_premium_mean <- mean(host_effect$premium, na.rm = TRUE)

# task 5 monte carlo forecast

# Set seed for reproductability
set.seed(42)
n_sim <- 1e7
z <- qnorm(0.975)

# Helper: convert a 95% CI to (mean, sd) assuming normality
ci_to_normal <- function(lower, upper) {
  m <- (lower + upper) / 2
  s <- (upper - m) / z
  list(mean = m, sd = s)
}

simulate_2028 <- function(m) {
  q1 <- q1_results |> filter(medal == m)
  q2 <- q2_results |> filter(medal == m)
  q3 <- q3_results |> filter(medal == m)

  base_p <- ci_to_normal(q1$lower_ci, q1$upper_ci)
  hf_p   <- ci_to_normal(q2$lower_ci, q2$upper_ci)  # pct change CI
  hp_p   <- ci_to_normal(q3$lower_ci, q3$upper_ci)

  baseline_samp    <- rnorm(n_sim, base_p$mean, base_p$sd)
  host_factor_samp <- 1 + rnorm(n_sim, hf_p$mean, hf_p$sd)  # factor = 1 + pct
  host_prob_samp   <- rnorm(n_sim, hp_p$mean, hp_p$sd)

  predicted <- baseline_samp * host_factor_samp + 22 * host_prob_samp

  tibble(
    medal       = m,
    baseline    = base_p$mean,
    host_factor = 1 + hf_p$mean,
    host_prob   = hp_p$mean,
    pred_lower  = quantile(predicted, 0.025),
    pred_median = quantile(predicted, 0.5),
    pred_upper  = quantile(predicted, 0.975)
  )
}

The Bottom Line

Team USA is positioned for what could be the most dominant Olympic performance in American history. Our forecast projects approximately 250 total medals, including roughly 120 gold, at the 2028 Los Angeles Summer Olympics. Even the conservative scenario - the lower bound of our 95% confidence range — comes in at 131 total medals and 49 golds, in line with USA’s strongest recent Olympics.

For sponsors, this is the rare combination of a home Games, a deep talent pool, and an expanded medal program tailored to American strengths.

The Three-Factor Model

We built our forecast around three components grounded in 60+ years of Olympic history:

1. USA Baseline. Across every Summer Olympics since 1960, Team USA has averaged 41 gold medals and 107 total medals per Games. This is the floor: USA shows up to every Olympics as the team to beat, regardless of host country or sport mix. Recent performances (46 golds in Rio, 39 in Tokyo, 40 in Paris) line up with this baseline.

2. The Host Nation Effect. Host countries consistently overperform their prior Olympics and usually by a wide margin. Recent examples make this concrete:

  • South Korea doubled its gold count when hosting Seoul 1988
  • China jumped from 32 golds in Athens 2004 to 48 in Beijing 2008
  • Great Britain won its highest medal count in a century at London 2012

The reasons are well documented: home crowds, specialized training facilities, and automatic qualification in every event combine to create a real boost. Our analysis estimates the home field advantage adds significantly to baseline performance, with the lower bound of statistical confidence still showing a positive effect.

3. The New Sport Effect. LA 2028 introduces six sports either new to the Olympics or returning after long absences: baseball, softball, lacrosse, cricket, flag football, and squash. These add 22 new medal events beyond the 2024 program. History shows host nations win medals in newly added sports about 57% of the time, and golds about 34% of the time. Team USA’s profile in these new events is exceptional:

  • Baseball & Softball: USA holds 4 of the 9 historical Olympic gold medals
  • Lacrosse: USA is the dominant historical performer
  • Flag Football: USA hhas been dominant in the professional flag football league - even beating a team made up of NFL stars
  • Cricket & Squash: Less of a historical strength, but still competitive entries
